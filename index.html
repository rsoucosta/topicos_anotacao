# Metadados do conjunto de dados
#
# Os dados analisados são compostos por 14 amostras paired-end:
# 7 amostras de tecido pulmonar normal
# 7 amostras tumorais de SCLC
#
# As etapas iniciais foram realizadas no Galaxy:
# 1. Importação dos dados SRA
# 2. Controle de qualidade com FastQC e MultiQC
# 3. Pré-processamento com Trim Galore!
# 4. Alinhamento ao genoma humano com HISAT2
# 5. Quantificação gênica com featureCounts
# 6. Análise de expressão diferencial com DESeq2
#
# A saída do DESeq2 foi utilizada como entrada neste script.

##############################################################
#################### Critérios de filtragem ##################
##############################################################

# Filtro inicial de genes diferencialmente expressos:
# FDR < 0,05
# |log2FoldChange| > 1
#
# Filtro para seleção de lncRNAs:
# gene_biotype classificado como lncRNA ou categorias relacionadas
# a RNAs longos não codificadores
#
# Filtro rígido para candidatos principais:
# FDR <= 0,01
# |log2FoldChange| >= 3
# baseMean >= 200

##############################################################
#################### STEP BY STEP ############################
##############################################################

# 1. Configuração inicial
#    a. Diretório de trabalho
#    b. Arquivo de entrada do DESeq2
#    c. Biotipos considerados como lncRNAs
#
# 2. Carregamento dos pacotes
#    a. biomaRt
#    b. AnnotationDbi
#    c. org.Hs.eg.db
#    d. dplyr
#    e. ggplot2
#    f. ggrepel
#
# 3. Leitura da saída do DESeq2
#    a. Importar o arquivo tabular gerado no Galaxy
#    b. Renomear as colunas do resultado
#    c. Converter ENTREZID para formato de texto
#
# 4. Filtragem inicial dos genes significativos
#    a. Remover genes sem padj
#    b. Genes com FDR < 0,05
#    c. Genes com |log2FoldChange| > 1
#    d. Tabela de genes diferencialmente expressos
#
# 5. Conversão dos identificadores gênicos
#    a. org.Hs.eg.db e AnnotationDbi
#    b. Converter Entrez Gene ID para Ensembl Gene ID
#    c. Remover identificadores Ensembl ausentes
#    d. Remover combinações duplicadas
#
# 6. Conexão com o banco Ensembl
#    a. Utilizar o pacote biomaRt
#    b. Selecionar o biomart de genes
#    c. Selecionar o dataset humano hsapiens_gene_ensembl
#
# 7. Anotação dos genes
#    a. Ensembl Gene ID
#    b. Símbolo gênico
#    c. Gene_biotype
#    d. Descrição do gene
#
# 8. União das tabelas
#    a. Unir a tabela do DESeq2 com a conversão Entrez-Ensembl
#    b. Tabela resultante com a anotação do biomaRt
#    c. Tabela anotada com símbolo gênico, biotipo e descrição
#
# 9. Remoção de duplicatas
#    a. Remover entradas sem identificador Ensembl
#    b. Ordenar os genes pelo menor FDR
#    c. Manter apenas uma entrada por identificador Ensembl
#
# 10. Seleção dos lncRNAs
#     a. Coluna gene_biotype
#     b. Genes classificados como lncRNA ou categorias relacionadas
#     c. Tabela final de lncRNAs diferencialmente expressos
#
# 11. Separação dos lncRNAs por direção da expressão
#     a. lncRNAs aumentados no SCLC
#     b. lncRNAs reduzidos no SCLC
#     c. Tabelas separadas para cada grupo
#
# 12. Seleção dos principais lncRNAs
#     a. Ordenar os lncRNAs pelo FDR
#     b. 20 lncRNAs mais significativos
#     c. 20 lncRNAs aumentados mais significativos
#     d. 20 lncRNAs reduzidos mais significativos

#     a. FDR <= 0,01
#     b. |log2FoldChange| >= 3
#     c. baseMean >= 200

#     e. Candidatos principais

################## 3. LEITURA DO DESeq2 ######################
##############################################################

# Importação do arquivo tabular gerado pelo DESeq2 no Galaxy
res <- read.delim(
  arquivo_deseq2,
  header = FALSE
)

# Nomeação das colunas da saída do DESeq2
colnames(res) <- c(
  "ENTREZID",
  "baseMean",
  "log2FoldChange",
  "lfcSE",
  "stat",
  "pvalue",
  "padj"
)

# Conversão do identificador Entrez para texto
res$ENTREZID <- as.character(res$ENTREZID)

##############################################################
############# 4. FILTRO INICIAL DE SIGNIFICÂNCIA #############
##############################################################

# Seleção dos genes diferencialmente expressos
res_sig_inicial <- res %>%
  filter(
    !is.na(padj),
    padj < 0.05,
    abs(log2FoldChange) > 1
  )

##############################################################
############ 5. CONVERSÃO ENTREZ ID PARA ENSEMBL #############
##############################################################

# Conversão dos identificadores Entrez Gene ID para Ensembl Gene ID
conv <- AnnotationDbi::select(
  org.Hs.eg.db,
  keys = res_sig_inicial$ENTREZID,
  columns = c("ENSEMBL"),
  keytype = "ENTREZID"
)
## 'select()' returned 1:many mapping between keys and columns
# Organização da tabela de conversão
conv <- conv %>%
  mutate(ENTREZID = as.character(ENTREZID)) %>%
  filter(!is.na(ENSEMBL)) %>%
  distinct()

##############################################################
################### 6. CONEXÃO COM ENSEMBL ###################
##############################################################

# Conexão com o banco Ensembl pelo pacote biomaRt
mart <- useEnsembl(
  biomart = "genes",
  dataset = "hsapiens_gene_ensembl"
)
## Ensembl site unresponsive, trying asia mirror
## Ensembl site unresponsive, trying useast mirror
##############################################################
###################### 7. ANOTAÇÃO GÊNICA ####################
##############################################################

# Recuperação do símbolo gênico, biotipo e descrição pelo Ensembl
annot <- getBM(
  attributes = c(
    "ensembl_gene_id",
    "external_gene_name",
    "gene_biotype",
    "description"
  ),
  filters = "ensembl_gene_id",
  values = unique(conv$ENSEMBL),
  mart = mart
)

##############################################################
###################### 8. UNIÃO DAS TABELAS ##################
##############################################################

# União da saída do DESeq2 com a conversão Entrez-Ensembl e anotação
res_sig_annot <- res_sig_inicial %>%
  left_join(conv, by = "ENTREZID") %>%
  left_join(
    annot,
    by = c("ENSEMBL" = "ensembl_gene_id")
  )

##############################################################
################### 9. REMOÇÃO DE DUPLICATAS #################
##############################################################

# Remoção de duplicatas, mantendo a entrada com menor FDR
res_sig_annot_unique <- res_sig_annot %>%
  filter(!is.na(ENSEMBL)) %>%
  arrange(padj) %>%
  distinct(ENSEMBL, .keep_all = TRUE)

##############################################################
#################### 10. SELEÇÃO DOS lncRNAs #################
##############################################################

# Seleção dos genes classificados como lncRNA ou categorias relacionadas
lncrnas <- res_sig_annot_unique %>%
  filter(
    gene_biotype %in% biotipos_lncRNA
  ) %>%
  distinct(ENSEMBL, .keep_all = TRUE)

##############################################################
############### 11. lncRNAs AUMENTADOS/REDUZIDOS #############
##############################################################

# lncRNAs com maior expressão no SCLC
lncrnas_up <- lncrnas %>%
  filter(log2FoldChange > 1)

# lncRNAs com menor expressão no SCLC
lncrnas_down <- lncrnas %>%
  filter(log2FoldChange < -1)

##############################################################
####################### 12. TOP 20 lncRNAs ###################
##############################################################

# Top 20 lncRNAs mais significativos
top20_lncRNAs <- lncrnas %>%
  arrange(padj) %>%
  dplyr::select(
    ENTREZID,
    ENSEMBL,
    external_gene_name,
    gene_biotype,
    baseMean,
    log2FoldChange,
    pvalue,
    padj,
    description
  ) %>%
  head(20)

  arrange(padj) %>%
  dplyr::select(
    ENTREZID,
    ENSEMBL,
    external_gene_name,
    gene_biotype,
    baseMean,
    log2FoldChange,
    pvalue,
    padj,
    description
  ) %>%
  head(20)

# Top 20 lncRNAs reduzidos no SCLC
top20_lncRNAs_down <- lncrnas_down %>%
  arrange(padj) %>%
  dplyr::select(
    ENTREZID,
    ENSEMBL,
    external_gene_name,
    gene_biotype,
    baseMean,
    log2FoldChange,
    pvalue,
    padj,
    description
  ) %>%
  head(20)

##############################################################
################## 13. lncRNAs DESTACADOS ####################
##############################################################

# Seleção dos lncRNAs discutidos no artigo
lncrnas_destaque <- lncrnas %>%
  filter(
    external_gene_name %in% genes_destaque
  ) %>%
  dplyr::select(
    external_gene_name,
    gene_biotype,
    baseMean,
    log2FoldChange,
    padj,
    description
  ) %>%
  arrange(desc(log2FoldChange))

##############################################################
################### 14. FILTRO RÍGIDO ########################
##############################################################

# Seleção dos candidatos principais
lncrnas_filtro_rigido <- lncrnas %>%
  filter(
    !is.na(padj),
    !is.na(baseMean),
    padj <= 0.01,
    abs(log2FoldChange) >= 3,
    baseMean >= 200
  )

# Candidatos aumentados no SCLC
lncrnas_filtro_rigido_up <- lncrnas_filtro_rigido %>%
  filter(log2FoldChange >= 3)

# Candidatos reduzidos no SCLC
lncrnas_filtro_rigido_down <- lncrnas_filtro_rigido %>%
  filter(log2FoldChange <= -3)

# Top 20 após filtro rígido
top20_lncRNAs_filtro_rigido <- lncrnas_filtro_rigido %>%
  arrange(padj) %>%
  dplyr::select(
    ENTREZID,
    ENSEMBL,
    external_gene_name,
    gene_biotype,
    baseMean,
    log2FoldChange,
    pvalue,
    padj,
    description
  ) %>%
  head(20)

############ 17. PREPARAÇÃO DO VOLCANO PLOT GERAL ############

# Preparação dos genes para o volcano plot geral
res_annot_plot <- res %>%
  mutate(
    padj_plot = ifelse(padj == 0, .Machine$double.xmin, padj),

    significance = case_when(
      padj < 0.05 & log2FoldChange > 1  ~ "Aumentado no SCLC",
      padj < 0.05 & log2FoldChange < -1 ~ "Reduzido no SCLC",
      TRUE ~ "Não significativo"
    ),

    neg_log10_padj = -log10(padj_plot),
    point_size = ifelse(significance == "Não significativo", 1, 1.8)
  ) %>%
  filter(
    !is.na(padj),
    !is.na(log2FoldChange),
    is.finite(neg_log10_padj)
  )

##############################################################
########## 18. lncRNAs DESTACADOS NO VOLCANO GERAL ###########
##############################################################

# Seleção dos principais lncRNAs reduzidos para aparecerem no gráfico geral
top_lncRNAs_down_geral <- lncrnas %>%
  filter(
    log2FoldChange < -1,
    !is.na(external_gene_name),
    external_gene_name != ""
  ) %>%
  arrange(padj) %>%
  distinct(external_gene_name, .keep_all = TRUE) %>%
  head(3)

# União dos candidatos do filtro rígido com os principais reduzidos
top_lncRNAs_geral <- bind_rows(
  lncrnas_filtro_rigido,
  top_lncRNAs_down_geral
) %>%
  mutate(
    padj_plot = ifelse(padj == 0, .Machine$double.xmin, padj)
  ) %>%
  distinct(external_gene_name, .keep_all = TRUE)

##############################################################
################## 19. VOLCANO PLOT GERAL ####################
##############################################################

volcano_geral <- ggplot(
  res_annot_plot,
  aes(
    x = log2FoldChange,
    y = neg_log10_padj,
    color = significance
  )
) +
  geom_point(aes(size = point_size), alpha = 0.65) +
  scale_size_identity() +
  scale_color_manual(
    values = c(
      "Aumentado no SCLC" = "#D73027",
      "Reduzido no SCLC" = "#4575B4",
      "Não significativo" = "grey80"
    )
  ) +
  geom_vline(
    xintercept = c(-1, 1),
    linetype = "dashed",
    color = "grey40",
    alpha = 0.5,
    linewidth = 0.3
  ) +
  geom_hline(
    yintercept = -log10(0.05),
    linetype = "dashed",
    color = "grey40",
    alpha = 0.5,
    linewidth = 0.3
  ) +
  geom_text_repel(
    data = top_lncRNAs_geral,
    aes(
      x = log2FoldChange,
      y = -log10(padj_plot),
      label = external_gene_name
    ),
    inherit.aes = FALSE,
    color = "black",
    size = 4.3,
    max.overlaps = 25,
    box.padding = 0.5,
    point.padding = 0.3,
    segment.color = "grey40"
  ) +
  coord_cartesian(xlim = c(-10, 10)) +
  labs(
    title = "Expressão diferencial de genes entre SCLC e tecido pulmonar normal",
    x = "log2(Fold Change)",
    y = "-log10(FDR)",
    color = "Expressão diferencial"
  ) +
  theme_classic(base_size = 15) +
  theme(
    plot.title = element_text(face = "bold"),
    legend.position = "right",
    legend.title = element_text(face = "bold")
  )
